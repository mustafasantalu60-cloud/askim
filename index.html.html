<script>
let mesajlar = [
"Seninle her şey daha güzel 💕",
"Gülüşün dünyamı aydınlatıyor ✨",
"İyi ki varsın hanımağam ❤️",
"Seni her geçen gün daha çok seviyorum 💘",
"Hayatımın en güzel parçasısın 🫶"
];

let index = 0;

function degistir() {
    index = (index + 1) % mesajlar.length;
    document.getElementById("yazi").innerText = mesajlar[index];
}

// Kalp yağmuru
function kalpOlustur() {
    let kalp = document.createElement("div");
    kalp.classList.add("heart");
    kalp.innerHTML = "💖";

    kalp.style.left = Math.random() * 100 + "vw";
    kalp.style.animationDuration = (Math.random() * 3 + 2) + "s";

    document.body.appendChild(kalp);

    setTimeout(() => {
        kalp.remove();
    }, 5000);
}

setInterval(kalpOlustur, 300);
